### 2. 题目显示 - load_question()

```python
def load_question(self, index):
    # 1. 停止自动跳转计时器（防止竞态）
    if hasattr(self, 'auto_next_timer'):
        self.auto_next_timer.stop()
    
    self.current_index = index
    q = self.questions[index]
    
    # 2. 更新进度显示
    self.lbl_progress.setText(f"进度: {index + 1}/{len(self.questions)}  [{q['type']}]")
    
    # 3. 显示题目（包含来源和等级信息）
    source = q.get('source_sheet', '未知来源')
    q_levels = ",".join(q.get('levels', []))
    source_html = f"<span style='color: #666; font-size: 14px;'>  (来源: {source} | 等级: {q_levels})</span>"
    self.lbl_question.setText(f"{index + 1}. {q['question']}{source_html}")
    
    # 4. 清空之前的选项和反馈
    self.lbl_feedback.setVisible(False)
    for i in reversed(range(self.options_layout.count())): 
        self.options_layout.itemAt(i).widget().setParent(None)

    # 5. 创建按钮组
    self.option_group = QButtonGroup(self)
    # 单选/判断题: exclusive=True, 多选题: exclusive=False
    self.option_group.setExclusive(q['type'] in ['单选题', '判断题'])
    
    # 6. 根据题型渲染选项
    if q['type'] in ['单选题', '多选题']:
        for key, val in sorted(q['options'].items()):
            if q['type'] == '单选题':
                btn = QRadioButton(f"{key}. {val}")
            else:
                btn = QCheckBox(f"{key}. {val}")
            
            btn.setStyleSheet("font-size: 18px; padding: 12px;")
            self.options_layout.addWidget(btn)
            self.option_group.addButton(btn)
    
    elif q['type'] == '判断题':
        # 判断题特殊处理: 简化为√/×
        use_key_as_answer = q['answer'] in q['options'].keys()
        
        for key, val in sorted(q['options'].items()):
            # 统一为√或×显示
            btn_text = val
            if val in ['正确', 'True', 'T', '对', '√']:
                btn_text = "√"
            elif val in ['错误', 'False', 'F', '错', '×']:
                btn_text = "×"
            
            btn = QRadioButton(btn_text)
            
            # 保存选项键（用于后续判分）
            if use_key_as_answer:
                btn.setProperty("option_key", key)  # 答案是"A"
            else:
                btn.setProperty("option_key", val)  # 答案是"√"
                
            btn.setStyleSheet("font-size: 24px; font-weight: bold; padding: 12px;")
            self.options_layout.addWidget(btn)
            self.option_group.addButton(btn)
    
    elif q['type'] == '简答题':
        hint_label = QLabel("(简答题请在心中作答，点击提交查看参考答案)")
        hint_label.setStyleSheet("color: #666; font-size: 12px;")
        self.options_layout.addWidget(hint_label)
    
    # 7. 连接自动判分信号（单选/判断题点击即判分）
    if q['type'] in ['单选题', '判断题']:
        for btn in self.option_group.buttons():
            btn.clicked.connect(self.check_current_answer)
    
    # 8. 恢复已答状态（如果已答过）
    if index in self.manager.user_answers:
        user_ans = self.manager.user_answers[index]
        self.restore_answer_state(q, user_ans)
    
    # 9. 更新导航按钮状态（高亮当前题、标记正误）
    for idx, nav_btn in enumerate(self.nav_buttons):
        # ... 根据答题状态设置颜色
    
    # 10. 延迟300ms自动读题
    if hasattr(self, 'auto_read_timer'):
        self.auto_read_timer.stop()
        self.auto_read_timer.start(300)
```

**关键点**:
1. **题型适配**: 不同题型渲染不同控件（Radio/CheckBox/Label）
2. **判断题简化**: 统一显示为√/×,提升用户体验
3. **自动判分**: 单选/判断题点击后立即调用 `check_current_answer()`
4. **状态恢复**: 已答题目状态持久化
5. **自动读题**: 加载题目后延迟300ms触发TTS

### 3. 答案检查 - check_current_answer()

```python
def check_current_answer(self):
    q = self.questions[self.current_index]
    user_ans = []
    
    # 1. 收集用户选择
    if q['type'] in ['单选题', '判断题']:
        for btn in self.option_group.buttons():
            if btn.isChecked():
                key = btn.property("option_key")  # 判断题用property
                if key:
                    user_ans.append(key)
                else:
                    user_ans.append(btn.text().split('.')[0])  # 单选题解析文本
    
    elif q['type'] == '多选题':
        for i in range(self.options_layout.count()):
            widget = self.options_layout.itemAt(i).widget()
            if isinstance(widget, QCheckBox) and widget.isChecked():
                user_ans.append(widget.text().split('.')[0])
    
    # 2. 拼接答案（多选题排序后拼接）
    final_ans = "".join(sorted(user_ans))
    
    # 3. 提交到ExamManager
    self.manager.submit_answer(self.current_index, final_ans)
    
    # 4. 判断正误
    correct_ans = q['answer']
    is_correct = final_ans == correct_ans
    
    # 5. 禁用选项（防止修改）
    if q['type'] in ['单选题', '判断题']:
        should_disable = True
        # 练习模式且答错: 允许重新作答
        if self.is_practice_mode and not is_correct:
            should_disable = False
        
        if should_disable:
            for btn in self.option_group.buttons():
                btn.setEnabled(False)
    
    # 6. 显示反馈
    if is_correct:
        self.lbl_feedback.setText("√ 回答正确！")
        self.lbl_feedback.setStyleSheet("color: #28a745; font-weight: bold;")
        self.lbl_feedback.setVisible(True)
        
        # 高亮正确选项（绿色）
        self._highlight_options(correct_ans, True)
        
        # ⭐ 1.5秒后自动跳转到下一题
        if self.current_index < len(self.questions) - 1:
            self.auto_next_timer.start(1500)
    
    else:
        self.lbl_feedback.setText(f"× 回答错误！ 正确答案: {correct_ans}")
        self.lbl_feedback.setStyleSheet("color: #dc3545; font-weight: bold;")
        self.lbl_feedback.setVisible(True)
        
        # 高亮正确选项（绿色）和错误选项（红色删除线）
        self._highlight_options(correct_ans, False, final_ans)
        
        # ⭐ 答错不自动跳转，停留查看反馈
    
    # 7. 更新导航按钮状态
    if is_correct:
        self.exam_states[self.current_index] = 'correct'
        self.nav_buttons[self.current_index].setStyleSheet("background-color: #28a745;")
    else:
        self.exam_states[self.current_index] = 'wrong'
        self.nav_buttons[self.current_index].setStyleSheet("background-color: #dc3545;")
    
    # 8. 保存练习进度
    if self.is_practice_mode:
        if is_correct:
            self.save_current_practice_progress(remove_wrong=True)
        else:
            self.save_current_practice_progress(add_wrong=True)
```

**核心逻辑**:
```
答对 → 绿色反馈 → 1.5秒后自动跳转
答错 → 红色反馈 → 停留查看正确答案
```

**跨平台实现**:
- Web: `setTimeout(() => nextQuestion(), 1500)`
- Android: `Handler().postDelayed({ nextQuestion() }, 1500)`
- iOS: `DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) { nextQuestion() }`

### 4. 选项高亮 - _highlight_options()

```python
def _highlight_options(self, correct_ans, is_correct, user_ans=None):
    """
    根据答题结果高亮选项
    correct_ans: 正确答案字符串
    is_correct: 用户是否答对
    user_ans: 用户答案字符串（仅答错时传入）
    """
    for i in range(self.options_layout.count()):
        widget = self.options_layout.itemAt(i).widget()
        if not isinstance(widget, (QRadioButton, QCheckBox)):
            continue
            
        text_key = widget.text().split('.')[0]  # 提取"A"、"B"等
        
        # 重置样式
        base_style = "font-size: 18px; padding: 12px;"
        if isinstance(widget, QRadioButton) and widget.text() in ['√', '×']:
            base_style = "font-size: 24px; font-weight: bold; padding: 12px;"
        
        # 高亮正确答案（绿色）
        if text_key in correct_ans:
            widget.setStyleSheet(base_style + "color: #28a745; font-weight: bold;")
        
        # 高亮错误答案（红色删除线）
        if not is_correct and user_ans and text_key in user_ans and text_key not in correct_ans:
            widget.setStyleSheet(base_style + "color: #dc3545; font-weight: bold; text-decoration: line-through;")
```

**视觉效果**:
- ✅ 正确选项: 绿色加粗
- ❌ 错误选项: 红色加粗 + 删除线
- ⚪ 未选选项: 默认样式

### 5. 自动跳转机制

```python
# 初始化计时器
self.auto_next_timer = QTimer(self)
self.auto_next_timer.setSingleShot(True)  # 单次触发
self.auto_next_timer.timeout.connect(lambda: self.load_question(self.current_index + 1))

# 答对时启动
if is_correct and self.current_index < len(self.questions) - 1:
    self.auto_next_timer.start(1500)  # 1500毫秒

# 切换题目时停止（防止竞态）
def load_question(self, index):
    if hasattr(self, 'auto_next_timer'):
        self.auto_next_timer.stop()
    # ...
```

**设计要点**:
1. **单次触发**: `setSingleShot(True)` 确保只跳转一次
2. **边界检查**: 最后一题不跳转
3. **竞态防护**: 加载新题时先停止计时器

### 6. 计时器管理

```python
def start_mock_exam(self):
    # 初始化45分钟倒计时
    self.remaining_time = 45 * 60  # 2700秒
    self.timer.start(1000)  # 每秒触发一次
    # ...

def update_timer(self):
    # 减少时间
    self.remaining_time -= 1
    
    # 显示格式化时间
    minutes = self.remaining_time // 60
    seconds = self.remaining_time % 60
    self.lbl_timer.setText(f"剩余时间: {minutes:02d}:{seconds:02d}")
    
    # 少于5分钟时变红
    if self.remaining_time < 300:
        self.lbl_timer.setStyleSheet("color: red;")
    
    # 时间耗尽自动交卷
    if self.remaining_time <= 0:
        self.timer.stop()
        self.finish_exam(auto=True)
```

### 7. 练习进度保存

```python
# 数据结构
self.practice_data = {
    "一级,二级": {  # 等级组合键
        "单选题": {
            "index": 15,  # 当前进度
            "wrong_indices": [3, 7, 12]  # 答错的题目索引
        },
        "多选题": {...}
    }
}

def save_current_practice_progress(self, add_wrong=False, remove_wrong=False):
    # 生成等级键
    level_key = ",".join(sorted(self.current_practice_levels))
    q_type = self.questions[0]['type']
    
    # 更新进度
    data = self.practice_data[level_key][q_type]
    data['index'] = self.current_index
    
    # 管理错题列表
    if add_wrong and self.current_index not in data['wrong_indices']:
        data['wrong_indices'].append(self.current_index)
    if remove_wrong and self.current_index in data['wrong_indices']:
        data['wrong_indices'].remove(self.current_index)
    
    # 持久化到JSON
    self.save_practice_progress()
```
