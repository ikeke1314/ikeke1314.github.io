# PC应用技术文档 - 跨平台适配指南

## 概述

本文档提供完整的跨平台适配方案,帮助开发者将PC应用移植到Web、Android、iOS等平台。

---

## 技术栈对比

### PC (当前实现)
- **语言**: Python 3.x
- **UI框架**: PyQt6
- **数据处理**: Pandas + openpyxl
- **TTS**: Windows SAPI (win32com)
- **存储**: JSON文件

### Web
- **语言**: JavaScript/TypeScript
- **UI框架**: React / Vue / Angular
- **数据处理**: SheetJS (xlsx.js)
- **TTS**: Web Speech API
- **存储**: LocalStorage / IndexedDB / 后端API

### Android
- **语言**: Kotlin / Java
- **UI框架**: Jetpack Compose / XML Layouts
- **数据处理**: Apache POI
- **TTS**: Android TTS API
- **存储**: Room Database / SharedPreferences

### iOS
- **语言**: Swift
- **UI框架**: SwiftUI / UIKit
- **数据处理**: CoreXLSX
- **TTS**: AVSpeechSynthesizer
- **存储**: CoreData / UserDefaults

    }

### 2. 加权随机抽题

#### 算法核心 (跨平台通用)
```javascript
// 伪代码,适用所有语言
function startExam(targetLevel, questionBank) {
    const levelWeights = {
        '一级': { '一级': 0.8, '二级': 0.2 },
        '二级': { '一级': 0.1, '二级': 0.7, '三级': 0.2 },
        // ...
    };
    
    const weights = levelWeights[targetLevel];
    const examConfig = { '单选题': 40, '多选题': 10, '判断题': 10, '简答题': 3 };
    
    let finalQuestions = [];
    let seenQuestions = new Set();
    
    for (let [qType, totalCount] of Object.entries(examConfig)) {
        let typeQuestions = [];
        
        // 计算各等级题目数量
        let countsPerLevel = {};
        let currentTotal = 0;
        
        for (let [level, weight] of Object.entries(weights)) {
            const count = Math.floor(totalCount * weight);
            countsPerLevel[level] = count;
            currentTotal += count;
        }
        
        // 分配余数
        const remainder = totalCount - currentTotal;
        if (remainder > 0) {
            countsPerLevel[targetLevel] = (countsPerLevel[targetLevel] || 0) + remainder;
        }
        
        // 从各等级抽题
        for (let [level, count] of Object.entries(countsPerLevel)) {
            if (count <= 0) continue;
            
            const availableQuestions = questionBank.filter(q => 
                q.type === qType && 
                q.levels.includes(level) &&
                !seenQuestions.has(q.question)
            );
            
            const selected = randomSample(availableQuestions, count);
            selected.forEach(q => seenQuestions.add(q.question));
            typeQuestions.push(...selected);
        }
        
        shuffle(typeQuestions);
        finalQuestions.push(...typeQuestions);
    }
    
    return finalQuestions;
}

function randomSample(array, count) {
    const shuffled = [...array].sort(() => Math.random() - 0.5);
    return shuffled.slice(0, Math.min(count, array.length));
}

function shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
    }
}
```

#### Web实现 (Web Speech API)
```javascript
function speak(text) {
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'zh-CN';  // 中文
    utterance.rate = 1.0;      // 语速
    
    // 选择中文语音
    const voices = speechSynthesis.getVoices();
    const chineseVoice = voices.find(v => v.lang.includes('zh'));
    if (chineseVoice) {
        utterance.voice = chineseVoice;
    }
    
    speechSynthesis.speak(utterance);
}

// 停止朗读
function stopSpeaking() {
    speechSynthesis.cancel();
}
```

#### Android实现
```kotlin
import android.speech.tts.TextToSpeech
import java.util.Locale

class TTSHelper(context: Context) {
    private var tts: TextToSpeech? = null
    
    init {
        tts = TextToSpeech(context) { status ->
            if (status == TextToSpeech.SUCCESS) {
                tts?.language = Locale.CHINESE
                tts?.setSpeechRate(1.0f)
            }
        }
    }
    
    fun speak(text: String) {
        tts?.speak(text, TextToSpeech.QUEUE_FLUSH, null, "utteranceId")
    }
    
    fun stop() {
        tts?.stop()
    }
    
    fun shutdown() {
        tts?.shutdown()
    }
}
```

### 4. UI自动跳转计时器

#### PC实现 (QTimer)
```python
from PyQt6.QtCore import QTimer

self.auto_next_timer = QTimer(self)
self.auto_next_timer.setSingleShot(True)
self.auto_next_timer.timeout.connect(self.next_question)

# 答对后1.5秒跳转
if is_correct:
    self.auto_next_timer.start(1500)

# 切换题目时停止
def load_question(self, index):
    if hasattr(self, 'auto_next_timer'):
        self.auto_next_timer.stop()
```

#### Web实现 (setTimeout)
```javascript
class ExamUI {
    constructor() {
        this.autoNextTimer = null;
    }
    
    checkAnswer(isCorrect) {
        if (isCorrect) {
            // 1.5秒后跳转
            this.autoNextTimer = setTimeout(() => {
                this.nextQuestion();
            }, 1500);
        }
    }
    
    loadQuestion(index) {
        // 清除之前的计时器
        if (this.autoNextTimer) {
            clearTimeout(this.autoNextTimer);
            this.autoNextTimer = null;
        }
        
        // 加载新题目...
    }
}
```

#### Android实现 (Handler)
```kotlin
import android.os.Handler
import android.os.Looper

class ExamActivity : AppCompatActivity() {
    private val handler = Handler(Looper.getMainLooper())
    private var autoNextRunnable: Runnable? = null
    
    fun checkAnswer(isCorrect: Boolean) {
        if (isCorrect) {
            // 1.5秒后跳转
            autoNextRunnable = Runnable {
                nextQuestion()
            }
            handler.postDelayed(autoNextRunnable!!, 1500)
        }
    }
    
    fun loadQuestion(index: Int) {
        // 取消之前的定时任务
        autoNextRunnable?.let {
            handler.removeCallbacks(it)
        }
        
        // 加载新题目...
    }
}
```

#### iOS实现 (DispatchQueue)
```swift
import Foundation

class ExamViewController: UIViewController {
    private var autoNextWorkItem: DispatchWorkItem?
    
    func checkAnswer(isCorrect: Bool) {
        if isCorrect {
            // 1.5秒后跳转
            let workItem = DispatchWorkItem { [weak self] in
                self?.nextQuestion()
            }
            autoNextWorkItem = workItem
            
            DispatchQueue.main.asyncAfter(deadline: .now() + 1.5, execute: workItem)
        }
    }
    
    func loadQuestion(index: Int) {
        // 取消之前的定时任务
        autoNextWorkItem?.cancel()
        autoNextWorkItem = nil
        
        // 加载新题目...
    }
}
```

```kotlin
@Database(entities = [ErrorQuestion::class, PracticeProgress::class], version = 1)
abstract class AppDatabase : RoomDatabase() {
    abstract fun errorDao(): ErrorQuestionDao
    abstract fun progressDao(): PracticeProgressDao
}

@Entity(tableName = "error_questions")
data class ErrorQuestion(
    @PrimaryKey(autoGenerate = true) val id: Long = 0,
    val question: String,
    val type: String,
    val correctAnswer: String,
    val userAnswer: String,
    val timestamp: Long
)

@Dao
interface ErrorQuestionDao {
    @Query("SELECT * FROM error_questions ORDER BY timestamp DESC")
    fun getAll(): Flow<List<ErrorQuestion>>
    
    @Insert
    suspend fun insert(error: ErrorQuestion)
    
    @Query("DELETE FROM error_questions")
    suspend fun deleteAll()
}
```

#### iOS实现 (CoreData)
```swift
import CoreData

class DataManager {
    let container = NSPersistentContainer(name: "ExamModel")
    
    init() {
        container.loadPersistentStores { _, error in
            if let error = error {
                fatalError("Core Data failed: \(error)")
            }
        }
    }
    
    func saveError(question: String, correctAnswer: String, userAnswer: String) {
        let context = container.viewContext
        let error = ErrorQuestion(context: context)
        error.question = question
        error.correctAnswer = correctAnswer
        error.userAnswer = userAnswer
        error.timestamp = Date()
        
        try? context.save()
    }
    
    func getAllErrors() -> [ErrorQuestion] {
        let request: NSFetchRequest<ErrorQuestion> = ErrorQuestion.fetchRequest()
        request.sortDescriptors = [NSSortDescriptor(key: "timestamp", ascending: false)]
        
        return (try? container.viewContext.fetch(request)) ?? []
    }
}
```

## 总结

本文档提供了全面的跨平台适配方案。核心算法(加权抽题、判分逻辑)是语言无关的,可直接翻译。UI和存储需根据平台特性选择合适的实现方式。

**关键要点**:
1. **数据结构保持一致**: JSON格式的题目和进度数据可跨平台共享
2. **算法逻辑通用**: 加权配置、随机抽题等核心逻辑适用所有语言
3. **UI适配平台特性**: 利用各平台原生组件提升用户体验
4. **渐进式迁移**: 优先实现核心功能,逐步添加高级特性
